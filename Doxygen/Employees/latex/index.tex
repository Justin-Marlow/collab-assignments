\begin{DoxyItemize}
\item \mbox{\hyperlink{classEmployee}{Employee}}\+: Base class
\item \mbox{\hyperlink{classSupervisor}{Supervisor}}\+: Derived class with bonus based on number of employees supervised
\item \mbox{\hyperlink{classOfficer}{Officer}}\+: Derived class with \textquotesingle{}evilness\textquotesingle{} factor in pay calculation
\end{DoxyItemize}\hypertarget{index_usage_sec}{}\doxysection{How to Use}\label{index_usage_sec}
